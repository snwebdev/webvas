  function load() {

    var a = JSON.parse(localStorage.getItem('data'));
    getObject(a);
    function getObject(a) {
      if (a instanceof Array) {
        for (let i = 0; i < a.length; i++) {
          getObject(a[i]);
        }
      } else if (a instanceof Object) {
        for (let prop in a) {
          // console.log('%c a','color:yellow',a,prop,a[prop]);
          var subObject = a[prop];
          if (subObject instanceof Object || subObject instanceof Array) {
            Object.keys(subObject).forEach(key=>{
              window[`${key.toUpperCase()}`] = new Object(`${subObject[`${key}`]}`);
              // console.log(`${subObject[`${key}`]}`);
            });
            getCont(prop);
          };
        };
      } else {
        var a = localStorage.getItem('data') ? JSON.parse(localStorage.getItem('data')) : [];
        var a = document.getElementById('container').children;
        var element = [];
        for ( var i = 0; i < a.length; i++ ) {
          var tag = a[i].tagName;
          var obj = {};
          obj[tag] = {
            "x" : 0,
            "y" : 0
          };
          element.push(obj);
        };
        localStorage.setItem("data", JSON.stringify(element));
      };
    };

  function drag(event) {
    // console.log("drag");
    if (active) {
      event.preventDefault();
      if (event.type === "touchmove") {
        item.currentX = event.touches[0].clientX - item.initialX;
        item.currentY = event.touches[0].clientY - item.initialY;
      } else {
        item.currentX = event.clientX - item.initialX;
        item.currentY = event.clientY - item.initialY;
      }

      item.xOffset = item.currentX;
      item.yOffset = item.currentY;

      setTranslate(item.currentX, item.currentY, item);
      function setTranslate(x, y, el) {
        el.style.transform = "translate3d(" + x + "px, " + y + "px, 0)";
      }
    }
  }


  function dragEnd(event) {
